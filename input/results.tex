\noindent \textbf { The first query with which a user is greeted is:}\\

\noindent function demoplot()\\
A = [1,2;3,4]\\
eig(A)\\
y = x = linspace(0,10);
[X,Y] = meshgrid(x,y);\\
mesh(X,Y,sin(X).*cos(Y).*X);\\
endfunction\\
